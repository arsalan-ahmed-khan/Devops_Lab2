<script>
const canvas = document.getElementById("game");
const ctx = canvas.getContext("2d");

function resize() {
    canvas.width = window.innerWidth;
    canvas.height = window.innerHeight;
}
window.addEventListener("resize", resize);
resize();

let gameOver = false;
let score = 0;
let frame = 0;

// Bird
const bird = {
    x: canvas.width * 0.2,
    y: canvas.height / 2,
    r: 16,
    vel: 0,
    gravity: 0.6,
    jump: -10
};

// Pipes
const pipes = [];
const pipeWidth = 80;
const gap = 180;
const speed = 3;

// Input
function flap() {
    if (!gameOver) {
        bird.vel = bird.jump;
    } else {
        reset();
    }
}
document.addEventListener("keydown", flap);
document.addEventListener("mousedown", flap);
document.addEventListener("touchstart", flap);

// Create pipe
function addPipe() {
    const top = Math.random() * (canvas.height - gap - 200) + 100;
    pipes.push({
        x: canvas.width,
        top,
        bottom: canvas.height - top - gap,
        passed: false
    });
}

// Draw background
function drawBackground() {
    // Sky
    ctx.fillStyle = "#70c5ce";
    ctx.fillRect(0, 0, canvas.width, canvas.height);

    // Ground
    ctx.fillStyle = "#ded895";
    ctx.fillRect(0, canvas.height - 80, canvas.width, 80);
}

// Draw bird (better look)
function drawBird() {
    ctx.save();
    ctx.translate(bird.x, bird.y);
    ctx.rotate(bird.vel * 0.04);

    // Body
    ctx.fillStyle = "#ffeb3b";
    ctx.beginPath();
    ctx.arc(0, 0, bird.r, 0, Math.PI * 2);
    ctx.fill();

    // Wing
    ctx.fillStyle = "#fbc02d";
    ctx.beginPath();
    ctx.arc(-5, 4, 8, 0, Math.PI * 2);
    ctx.fill();

    // Eye
    ctx.fillStyle = "white";
    ctx.beginPath();
    ctx.arc(6, -5, 4, 0, Math.PI * 2);
    ctx.fill();

    ctx.fillStyle = "black";
    ctx.beginPath();
    ctx.arc(7, -5, 2, 0, Math.PI * 2);
    ctx.fill();

    ctx.restore();
}

// Draw pipes
function drawPipes() {
    ctx.fillStyle = "#2ecc71";
    pipes.forEach(p => {
        ctx.fillRect(p.x, 0, pipeWidth, p.top);
        ctx.fillRect(p.x, canvas.height - p.bottom - 80, pipeWidth, p.bottom);
    });
}

// Collision
function hitPipe(p) {
    return (
        bird.x + bird.r > p.x &&
        bird.x - bird.r < p.x + pipeWidth &&
        (bird.y - bird.r < p.top ||
         bird.y + bird.r > canvas.height - p.bottom - 80)
    );
}

// Update
function update() {
    drawBackground();

    bird.vel += bird.gravity;
    bird.y += bird.vel;

    if (frame % 110 === 0) addPipe();

    pipes.forEach(p => {
        p.x -= speed;

        if (!p.passed && p.x + pipeWidth < bird.x) {
            score++;
            document.getElementById("score").innerText = score;
            p.passed = true;
        }

        if (hitPipe(p)) gameOver = true;
    });

    drawPipes();
    drawBird();

    if (bird.y + bird.r > canvas.height - 80 || bird.y - bird.r < 0) {
        gameOver = true;
    }

    if (gameOver) {
        ctx.fillStyle = "rgba(0,0,0,0.5)";
        ctx.fillRect(0,0,canvas.width,canvas.height);
        ctx.fillStyle = "white";
        ctx.font = "48px Arial";
        ctx.fillText("GAME OVER", canvas.width/2 - 140, canvas.height/2);
        return;
    }

    frame++;
    requestAnimationFrame(update);
}

// Reset
function reset() {
    pipes.length = 0;
    bird.y = canvas.height / 2;
    bird.vel = 0;
    score = 0;
    frame = 0;
    gameOver = false;
    document.getElementById("score").innerText = "0";
    update();
}

update();
</script>
